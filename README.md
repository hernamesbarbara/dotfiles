# You'll need to symlink all your dotfiles to your home directory

```bash
ln -s ~/dotfiles/aliases ~/.aliases
ln -s ~/dotfiles/bash_profile ~/.bash_profile
ln -s ~/dotfiles/greetoftheday ~/greetoftheday
ln -s ~/dotfiles/inputrc ~/.inputrc
ln -s ~/dotfiles/bash_prompt ~/.bash_prompt
ln -s ~/dotfiles/pgpass ~/.pgpass
ln -s ~/dotfiles/vimrc ~/.vimrc
ln -s ~/dotfiles/bashrc ~/.bashrc
ln -s ~/dotfiles/profile ~/.profile
ln -s ~/dotfiles/vim ~/.vim
ln -s ~/dotfiles/gitconfig ~/.gitconfig
ln -s ~/dotfiles/gitignore_global ~/.gitignore_global
ln -s ~/dotfiles/ackrc ~/.ackrc
ln -s ~/dotfiles/gibo-completion.bash ~/.gibo-completion.bash
ln -s ~/dotfiles/gist ~/.gist
ln -s ~/dotfiles/gitignore-boilerplates ~/.gitignore-boilerplates
```
